	[]
		范围
			/[a-z]/ 			所有小写字母
			/[A-Z]/ 			所有大写字母
			/[0-9]/ 			所有数字
			/[a-zA-Z0-9]/ 		所有的英文字母和数字

		任选一个
			/[abc]/ 			a b c
			/[ac]/ 				a c
			/a[abc]c/ 			aac 	abc 	acc

		排除
			/[^a-z]/ 			除了小写字母
			/[^A-Z]/ 			除了大写字母
			/[^0-9]/ 			除了数字
			/[^a-zA-Z0-9]/ 		除了英文和数字


		/[13-59]/ 				1 3 4 5 9

		/(ab)+/
		abababababab 			√

	-----------------------------------------------
	量词
		{n} 			有n个
		{n,m} 			最少有n个，最多有m个。
		{n,} 			最少有n个，最多不限

		? 	{0,1} 		出现一次或者不出现
		* 	{0,} 		不出现或者出现(出现多少不管)
		+ 	{1,} 		最少一次，最多不限

	------------------------------------------------
	转义
		\d 				所有数字
		\w 				所有英文字母和数字和_
		\s 				所有的空白符号

		\D 				除了数字
		\W 				除了英文和数字和_
		\S 				除了空白符号

		. 				所有的字符

		\. 		正常的字符串.

 	固定电话
 		区号 		3-4位数字
 			第一位必须是0
 			第二位必须不是0

 			0[1-9]\d{1,2}

 		- 			\-

 		座机号码
 			7-8位数字
 			第一位必须不是0
 			[1-9]\d{6,7}

 		/^0[1-9]\d{1,2}\-[1-9]\d{6,7}$/

 	年龄校验:
 		婚恋网站
 			18-100

 			18-19 		1[89]
 			20-99 		[2-9]\d
 			100 		100

 		/^((1[89])|([2-9]\d)|(100))$/

 	手机号
 		11位数字
 		第一位一定是1
 		第二位 	3 4 5 7 8

 		/^1[34578]\d{9}$/

 	身份证号
 		15位或18位数字
 		第一位必须不是0
 		最后一位有可能是x
 		/^[1-9]\d{13}(([0-9xX])|(\d{3}[0-9xX]))$/
